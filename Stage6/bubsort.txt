{

void swap()
{
	temp = foo[$1]
	foo[$1] = foo[$2]
	foo[$2] = temp
}

for(i = 0; i < len(foo); i++)
{
	for(j = i+1; j < len(foo); j++)
	{
		if(foo[i] >= foo[j])
		{
			swap(i,j)
		}
	}
}

{
	print foo[i]," "
