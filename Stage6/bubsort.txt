//a function to sort input array

print "enter array size: "
arr foo[read()]

for(i = 0; i < len(foo); i++)
{
	foo[i] = read()
}

void swap(i,j)
{
	temp = foo[i]
	foo[i] = foo[j]
	foo[j] = temp
}

for(i = 0; i < len(foo); i++)
{
	for(j = i+1; j < len(foo); j++)
	{
		if(foo[i] >= foo[j])
		{
			swap(i,j)
		}
	}
}

for(i = 0; i < len(foo); i++)
{
	print foo[i]," "
}

print "\n"
